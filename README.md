import cv2
import numpy as np
image = cv2.imread('p4.jpg')
cv2.imshow('Old', image)

cv2.imshow('Gray', gray)
cv2.imwrite('sample.jpg',gray)
cv2.waitKey(0)
cv2.destroyAllWindows()
